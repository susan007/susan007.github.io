# Es6 模板字符串

# 背景

Es6引入了一种新型的字符串字面量语法，叫做模板字符串。使用反撇号括起来的字符串。

# 基础知识

function test(user, action) {
  if(!user) {
      alert('  `用户$(user.name)非法闯入系统执行${action}操作` ')
  }
}
  • ${user.name}和${action}称之为模板占位符。
  • 在模板字符串中使用特殊符号需要转义。比如：` $ {

# 使用实例

//使用反撇号引用文字
let info = `噢，我亲爱的土拨鼠`
//反撇号中打印反撇号
let info = `噢，我亲爱的\`土拨鼠`
//空格 换行 缩进都会被保留
let info = `噢，
我亲爱的     土
   拨鼠`
//嵌入变量
let name = '张全蛋'
let info = `噢，我亲爱的土拨鼠${name}先生`
//嵌入表达式
let name = '张全蛋'
let age = 28
let info = `噢，我亲爱的土拨鼠${name}先生，过完生日您就${age+1}岁了`

实例二(搬运自阮一峰老师)

//将各个参数按照原来的位置拼接回去
let total = 30;
let msg = passthru`The total is ${total} (${total*1.05} with tax)`;

function passthru(literals, ...values) {
  let output = "";
  let index;
  for (index = 0; index < values.length; index++) {
    output += literals[index] + values[index];
  }

  output += literals[index]
  return output;
}

标签模板过滤html(搬运自阮一峰老师)

let message =
  SaferHTML`<p>${sender} has sent you a message.</p>`;

function SaferHTML(templateData) {
  let s = templateData[0];
  for (let i = 1; i < arguments.length; i++) {
    let arg = String(arguments[i]);

    // Escape special characters in the substitution.
    s += arg.replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;");

    // Don't escape special characters in the template.
    s += templateData[i];
  }
  return s;
}
message //<p>&lt;script&gt;alert("abc")&lt;/script&gt; has sent you a message.
