/* components/card-coupon-mine/card-coupon-mine.wxss */

.sawtooth {
  /* 相对定位，方便让before和after伪元素绝对定位偏移 */
  position: relative;
  /* 把超出div的部分隐藏起来 */
  overflow: hidden;
  height: 6rem;
  line-height: 6rem;
  background-color: hotpink;
  margin: 20rpx 20rpx;
  text-align: center;
  color: #fff;
  font-size: 2rem;
}

.sawtooth_tip {
  color: #aaa;
  font-size: small;
}

.sawtooth:before, .sawtooth:after {
  content: ' ';
  width: 0;
  height: 100%;
  /* 绝对定位进行偏移 */
  position: absolute;
  top: 10px;
}

.sawtooth:before {
  /* 圆点型的border */
  border-right: 10px dotted #f4f4f4;
  /* 偏移一个半径，让圆点的一半覆盖div */
  left: -5px;
}

.sawtooth:after {
  /* 圆点型的border */
  border-left: 10px dotted #f4f4f4;
  /* 偏移一个半径，让圆点的一半覆盖div */
  right: -5px;
}
